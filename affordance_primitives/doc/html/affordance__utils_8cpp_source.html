38 PoseStamped getPoseInATFrame(const std::string& root_frame_name, const PoseStamped& root_frame_pose,
39  const PoseStamped& incoming_frame)
40 {
41  // If the frame is already in the AT root, just return it
42  if (incoming_frame.header.frame_id == root_frame_name)
43  {
44  return incoming_frame;
45  }
46 
47  // Otherwise, try to convert it to the AT root frame
48  Eigen::Isometry3d tf_AT_root_to_pose = convertPoseToNewFrame(root_frame_pose, incoming_frame);
49 
50  PoseStamped output;
51  output.header.frame_id = root_frame_name;
52  output.pose = tf2::toMsg(tf_AT_root_to_pose);
53  return output;
54 }

56 TwistStamped getTwistFromPoses(const PoseStamped& start_pose, const PoseStamped& end_pose)
57 {
58  // Can't do it if they are in different frames
59  if (start_pose.header.frame_id != end_pose.header.frame_id)
60  {
61  throw std::runtime_error("Poses must be in same frame");
62  }
63 
64  // Get TF start -> end
65  Eigen::Isometry3d eigen_start_pose, eigen_end_pose;
66  Eigen::fromMsg(start_pose.pose, eigen_start_pose);
67  Eigen::fromMsg(end_pose.pose, eigen_end_pose);
68 
69  // (start -> header_frame) * (header_frame -> end)
70  Eigen::Isometry3d diff_pose = eigen_start_pose.inverse() * eigen_end_pose;
71 
72  // Use Axis Angle representation to get the Twist rotational values
73  Eigen::AngleAxisd axis_angle(diff_pose.linear());
74 
75  // Populate the output
76  TwistStamped output;
77  output.header.frame_id = start_pose.header.frame_id;
78  tf2::toMsg(diff_pose.translation(), output.twist.linear);
79  tf2::toMsg(axis_angle.angle() * axis_angle.axis(), output.twist.angular);
80 
81  return output;
82 }

84 Eigen::Isometry3d convertPoseToNewFrame(const PoseStamped& new_base_frame, const PoseStamped& transformed_pose)
85 {
86  if (new_base_frame.header.frame_id != transformed_pose.header.frame_id)
87  {
88  throw std::runtime_error("Poses must be in same frame");
89  }
90 
91  // Convert to Eigen
92  Eigen::Isometry3d tf_root_to_new_base_frame, tf_root_to_transformed_pose;
93  tf2::fromMsg(new_base_frame.pose, tf_root_to_new_base_frame);
94  tf2::fromMsg(transformed_pose.pose, tf_root_to_transformed_pose);
95 
96  // Do (new -> root) * (root -> input) = (new -> input)
97  return tf_root_to_new_base_frame.inverse() * tf_root_to_transformed_pose;
98 }

100 ScrewStamped transformScrew(const ScrewStamped& input_screw, const TransformStamped& transform)
101 {
102  if (input_screw.header.frame_id == transform.child_frame_id)
103  {
104  // Already in the correct frame
105  return input_screw;
106  }
107  else if (input_screw.header.frame_id != transform.header.frame_id)
108  {
109  throw std::runtime_error("Cannot transform screw: transform frame does not match incoming screw frame");
110  }
111 
112  ScrewStamped output;
113  output.header.frame_id = transform.child_frame_id;
114 
115  // Convert to Eigen types
116  auto tf_new_to_old = (tf2::transformToEigen(transform.transform)).inverse();
117  Eigen::Vector3d screw_axis, screw_origin;
118  tf2::fromMsg(input_screw.axis, screw_axis);
119  tf2::fromMsg(input_screw.origin, screw_origin);
120 
121  // Perform the transform
122  Eigen::Vector3d rotated_axis = tf_new_to_old.linear() * screw_axis;
123  Eigen::Vector3d transformed_origin = tf_new_to_old.translation() + tf_new_to_old.linear() * screw_origin;
124 
125  // Convert back from Eigen types
126  output.origin = tf2::toMsg(transformed_origin);
127  tf2::toMsg(rotated_axis, output.axis);
128  return output;
129 }

131 Eigen::Matrix3d getSkewSymmetricMatrix(const Eigen::Vector3d& vec)
132 {
133  Eigen::Matrix3d output;
134  output.setZero();
135  output(0, 1) = -vec(2);
136  output(1, 0) = vec(2);
137  output(0, 2) = vec(1);
138  output(2, 0) = -vec(1);
139  output(1, 2) = -vec(0);
140  output(2, 1) = vec(0);
141 
142  return output;
143 }

145 Eigen::MatrixXd getAdjointMatrix(const Eigen::Isometry3d& transform)
146 {
147  Eigen::MatrixXd adjoint(6, 6);
148 
149  // Block matrix with
150  // |R [p]R|
151  // |0 R |
152  adjoint.block<3, 3>(0, 0) = transform.linear();
153  adjoint.block<3, 3>(3, 0).setZero();
154  adjoint.block<3, 3>(0, 3) = getSkewSymmetricMatrix(transform.translation()) * transform.linear();
155  adjoint.block<3, 3>(3, 3) = transform.linear();
156 
157  return adjoint;
158 }

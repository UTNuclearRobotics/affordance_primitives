8 void enforceAPLimits(const APRobotParameter& params, AffordancePrimitiveFeedback& feedback, const double epsilon)
9 {
10  // Convert EE velocity twist and limits to Eigen
11  Eigen::Matrix<double, 6, 1> twist_limit = CartesianFloatToVector(params.max_ee_velocity);
12  Eigen::Matrix<double, 6, 1> twist_requested;
13  tf2::fromMsg(feedback.moving_frame_twist.twist, twist_requested);
14 
15  // Apply twist limit
16  double scale = 1.0;
17  for (size_t i = 0; i < 6; i++)
18  {
19  if (fabs(twist_limit[i]) > epsilon && fabs(twist_requested[i] / twist_limit[i]) > 1)
20  {
21  scale = std::max(scale, fabs(twist_requested[i] / twist_limit[i]));
22  }
23  }
24  twist_requested /= scale;
25 
26  // Convert wrench and limits to Eigen
27  Eigen::Matrix<double, 6, 1> wrench_limit = CartesianFloatToVector(params.max_wrench);
28  Eigen::Matrix<double, 6, 1> wrench_requested = WrenchToVector(feedback.expected_wrench.wrench);
29 
30  // Check for limit along each dimension
31  for (size_t i = 0; i < 6; i++)
32  {
33  if (fabs(wrench_limit[i]) > epsilon)
34  {
35  if (wrench_requested[i] < -1 * wrench_limit[i] || wrench_requested[i] > wrench_limit[i])
36  {
37  wrench_requested[i] = copysign(wrench_limit[i], wrench_requested[i]);
38  }
39  }
40  }
41 
42  // Check for total magnitude
43  if (fabs(params.max_force) > epsilon)
44  {
45  const double scale_force = wrench_requested.head(3).norm() / params.max_force;
46  if (scale_force > 1.0)
47  {
48  wrench_requested.head(3) /= scale_force;
49  }
50  }
51  if (fabs(params.max_torque) > epsilon)
52  {
53  const double scale_torque = wrench_requested.tail(3).norm() / params.max_torque;
54  if (scale_torque > 1.0)
55  {
56  wrench_requested.tail(3) /= scale_torque;
57  }
58  }
59 
60  // Convert back to message
61  feedback.moving_frame_twist.twist = tf2::toMsg(twist_requested);
62  feedback.expected_wrench.wrench = VectorToWrench(wrench_requested);
63 }

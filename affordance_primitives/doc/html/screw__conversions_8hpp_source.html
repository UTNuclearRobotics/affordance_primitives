46 inline Eigen::Matrix3d getSkewSymmetricMatrix(const Eigen::Vector3d& vec)
47 {
48  Eigen::Matrix3d output;
49  output.setZero();
50  output(0, 1) = -vec(2);
51  output(1, 0) = vec(2);
52  output(0, 2) = vec(1);
53  output(2, 0) = -vec(1);
54  output(1, 2) = -vec(0);
55  output(2, 1) = vec(0);
56 
57  return output;
58 }

69 inline Eigen::MatrixXd getAdjointMatrix(const Eigen::Isometry3d& transform)
70 {
71  Eigen::MatrixXd adjoint(6, 6);
72 
73  // Block matrix with
74  // |R 0|
75  // |[p]R R|
76  adjoint.block<3, 3>(0, 0) = transform.linear();
77  adjoint.block<3, 3>(0, 3).setZero();
78  adjoint.block<3, 3>(3, 0) = getSkewSymmetricMatrix(transform.translation()) * transform.linear();
79  adjoint.block<3, 3>(3, 3) = transform.linear();
80 
81  return adjoint;
82 }

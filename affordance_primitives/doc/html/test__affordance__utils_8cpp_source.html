70 TEST(AffordanceUtils, getPoseInATFrame)
71 {
72  affordance_primitives::PoseStamped AT_base_pose, other_pose, result;
73 
74  AT_base_pose.header.frame_id = PLANNING_FRAME_NAME;
75  AT_base_pose.pose.position.x = 1;
76  AT_base_pose.pose.position.y = 1;
77  AT_base_pose.pose.position.z = 1;
78 
79  // Orientation is a 90-degree roll about the X-axis
80  other_pose.pose.orientation.x = 0.5 * sqrt(2);
81  other_pose.pose.orientation.y = 0;
82  other_pose.pose.orientation.z = 0;
83  other_pose.pose.orientation.w = 0.5 * sqrt(2);
84 
85  // First, if the other_pose is in the AT frame already, it should return as-is
86  other_pose.header.frame_id = ROOT_FRAME_NAME;
87  ASSERT_NO_THROW(result = affordance_primitives::getPoseInATFrame(ROOT_FRAME_NAME, AT_base_pose, other_pose));
88  checkPoint(result.pose.position, 0, 0, 0);
89  checkQuaternion(result.pose.orientation, 0.5 * sqrt(2), 0, 0, 0.5 * sqrt(2));
90 
91  // Second, if the other_pose is also in the planning frame, a conversion should happen
92  other_pose.header.frame_id = PLANNING_FRAME_NAME;
93  ASSERT_NO_THROW(result = affordance_primitives::getPoseInATFrame(ROOT_FRAME_NAME, AT_base_pose, other_pose));
94 
95  // The result should be AT frame -> input pose
96  checkPoint(result.pose.position, -1, -1, -1);
97  checkQuaternion(result.pose.orientation, 0.5 * sqrt(2), 0, 0, 0.5 * sqrt(2));
98 
99  // Finally, it should throw if it cannot convert the pose
100  other_pose.header.frame_id = "some_other_frame";
101  EXPECT_THROW(result = affordance_primitives::getPoseInATFrame(ROOT_FRAME_NAME, AT_base_pose, other_pose),
102  std::runtime_error);
103 }

134 TEST(AffordanceUtils, convertPoseToNewFrame)
135 {
136  affordance_primitives::PoseStamped grasp_pose, screw_origin_pose;
137  Eigen::Isometry3d result;
138 
139  // First make sure we throw if the frames do not match
140  grasp_pose.header.frame_id = ROOT_FRAME_NAME;
141  screw_origin_pose.header.frame_id = PLANNING_FRAME_NAME;
142  EXPECT_THROW(result = affordance_primitives::convertPoseToNewFrame(grasp_pose, screw_origin_pose), std::runtime_error);
143 
144  // If the new base is identity, we should get back what we put in
145  screw_origin_pose.header.frame_id = ROOT_FRAME_NAME;
146  screw_origin_pose.pose.position.x = 1;
147  screw_origin_pose.pose.position.y = 1;
148  screw_origin_pose.pose.position.z = 0;
149 
150  // Orientation is a 90-degree yaw about the Z-axis
151  screw_origin_pose.pose.orientation.x = 0;
152  screw_origin_pose.pose.orientation.y = 0;
153  screw_origin_pose.pose.orientation.z = 0.5 * sqrt(2);
154  screw_origin_pose.pose.orientation.w = 0.5 * sqrt(2);
155 
156  // Check the result
157  ASSERT_NO_THROW(result = affordance_primitives::convertPoseToNewFrame(grasp_pose, screw_origin_pose));
158  checkVector(result.translation(), 1, 1, 0);
159  checkQuaternion(Eigen::Quaterniond(result.linear()), 0, 0, 0.5 * sqrt(2), 0.5 * sqrt(2));
160 
161  // Now check harder case
162  grasp_pose.pose.position.x = 1;
163  grasp_pose.pose.position.y = 0;
164  grasp_pose.pose.position.z = 0;
165 
166  // Check the result
167  ASSERT_NO_THROW(result = affordance_primitives::convertPoseToNewFrame(grasp_pose, screw_origin_pose));
168  checkVector(result.translation(), 0, 1, 0);
169  checkQuaternion(Eigen::Quaterniond(result.linear()), 0, 0, 0.5 * sqrt(2), 0.5 * sqrt(2));
170 }

172 TEST(AffordanceUtils, transformScrew)
173 {
174  affordance_primitives::ScrewStamped input_screw_msg, output_screw_msg;
175  affordance_primitives::TransformStamped tf_msg;
176 
177  // Set up TF: translate +3 on X axis, rotate +90 on Z
178  tf_msg.header.frame_id = PLANNING_FRAME_NAME;
179  tf_msg.child_frame_id = "new_screw_frame";
180  tf_msg.transform.translation.x = 3.0;
181  tf_msg.transform.rotation.x = 0;
182  tf_msg.transform.rotation.y = 0;
183  tf_msg.transform.rotation.z = 0.5 * sqrt(2);
184  tf_msg.transform.rotation.w = 0.5 * sqrt(2);
185 
186  // Set up screw axis
187  input_screw_msg.origin.y = 2.0;
188  input_screw_msg.axis.x = 1.0;
189  input_screw_msg.is_pure_translation = false;
190 
191  // If the screw is already in new frame, no transforms should happen
192  input_screw_msg.header.frame_id = "new_screw_frame";
193  ASSERT_NO_THROW(output_screw_msg = affordance_primitives::transformScrew(input_screw_msg, tf_msg));
194  checkVector(output_screw_msg.axis, 1, 0, 0);
195  checkPoint(output_screw_msg.origin, 0, 2, 0);
196  EXPECT_EQ(output_screw_msg.header.frame_id, input_screw_msg.header.frame_id);
197 
198  // If the TF doesn't have the frames we need, we can't do the TF
199  input_screw_msg.header.frame_id = "some_random_frame";
200  ASSERT_THROW(output_screw_msg = affordance_primitives::transformScrew(input_screw_msg, tf_msg), std::runtime_error);
201 
202  // Otherwise, the transform should work
203  input_screw_msg.header.frame_id = PLANNING_FRAME_NAME;
204  ASSERT_NO_THROW(output_screw_msg = affordance_primitives::transformScrew(input_screw_msg, tf_msg));
205 
206  // From the new frame, the axis should be -y and the origin should be (2, 3, 0)
207  EXPECT_EQ(output_screw_msg.header.frame_id, tf_msg.child_frame_id);
208  checkPoint(output_screw_msg.origin, 2, 3, 0);
209  checkVector(output_screw_msg.axis, 0, -1, 0);
210 }

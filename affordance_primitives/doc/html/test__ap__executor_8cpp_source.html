75 TEST_F(APExecutorTest, enforce_individual_limits)

79  feedback.expected_wrench.wrench.torque.x = -5.5;
80  feedback.expected_wrench.wrench.torque.y = 5.5;
81  feedback.expected_wrench.wrench.force.z = -51;

93  EXPECT_NEAR(feedback.expected_wrench.wrench.force.x, 1, EPSILON);
94  EXPECT_NEAR(feedback.moving_frame_twist.twist.angular.x, -0.1, EPSILON);
95  EXPECT_NEAR(feedback.moving_frame_twist.twist.linear.y, 0.2, EPSILON);
96  EXPECT_NEAR(feedback.moving_frame_twist.twist.linear.z, 0.0, EPSILON);

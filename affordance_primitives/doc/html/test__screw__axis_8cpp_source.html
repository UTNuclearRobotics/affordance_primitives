64 TEST(ScrewAxis, calculate_linear_component)
65 {
66  Eigen::Vector3d z_axis(0, 0, 1);
67  Eigen::Vector3d q_vector(-1, 0, 0);
68  double pitch = 0;
69 
70  Eigen::Vector3d linear_velocity = affordance_primitives::calculateLinearVelocity(z_axis, q_vector, pitch);
71 
72  // This should result in positive y velocity only
73  checkVector(linear_velocity, 0.0, 1.0, 0.0);
74 
75  // Changing the q-vector should result in nonzero X-Y velocity
76  q_vector.x() = -1 * sqrt(2) / 2;
77  q_vector.y() = sqrt(2) / 2;
78  linear_velocity = affordance_primitives::calculateLinearVelocity(z_axis, q_vector, pitch);
79  EXPECT_NEAR(linear_velocity.y(), linear_velocity.x(), EPSILON);
80  double norm = sqrt(pow(linear_velocity.y(), 2) + pow(linear_velocity.x(), 2));
81  EXPECT_NEAR(norm, 1.0, EPSILON);
82 
83  // Changing pitch should give us z velocity
84  pitch = 1;
85  linear_velocity = affordance_primitives::calculateLinearVelocity(z_axis, q_vector, pitch);
86  EXPECT_NEAR(linear_velocity.z(), 1.0, EPSILON);
87 }

247 TEST(ScrewAxis, get_twist)
248 {
249  affordance_primitives::ScrewAxis screw_axis(MOVING_FRAME_NAME);
250 
251  // Origin at [1,0,0], axis of [1,1,0]
252  affordance_primitives::Pose first_pose;
253  first_pose.position.x = 1.0;
254  Eigen::Vector3d wonky_axis(1, 1, 0);
255 
256  // The (instantaneous) velocity here is just Z axis
257  ASSERT_TRUE(screw_axis.setScrewAxis(first_pose, wonky_axis));
258  auto twist_msg = screw_axis.getTwist(0.5);
259  checkVector(twist_msg.twist.linear, 0.0, 0.0, 0.25 * sqrt(2)); // 0.5 vel * sqrt(2)/2 distance
260 
261  // The rotation is crazy, but is Roll and Pitch: 0.5 vel * sqrt(2)/2 distance
262  checkVector(twist_msg.twist.angular, 0.25 * sqrt(2), 0.25 * sqrt(2), 0.0);
263 
264  // Now check linear only motion
265  affordance_primitives::ScrewAxis screw_axis_linear(MOVING_FRAME_NAME, true);
266  ASSERT_TRUE(screw_axis_linear.setScrewAxis(first_pose, wonky_axis));
267  twist_msg = screw_axis_linear.getTwist(0.5);
268 
269  // Should be along axis with 0.5 vel * norm([1,1,0]) velocity in X-Y
270  checkVector(twist_msg.twist.linear, 0.25 * sqrt(2), 0.25 * sqrt(2), 0.0);
271 
272  // No rotation expected
273  checkVector(twist_msg.twist.angular, 0.0, 0.0, 0.0);
274 }

276 TEST(ScrewAxis, get_transform)
277 {
278  affordance_primitives::ScrewAxis screw_axis(MOVING_FRAME_NAME);
279 
280  // Origin at [1,0,0], axis of [0, 0, 1]
281  affordance_primitives::Pose first_pose;
282  first_pose.position.x = 1.0;
283  Eigen::Vector3d z_axis(0, 0, 1);
284 
285  // Will test a 90-degree rotation
286  ASSERT_TRUE(screw_axis.setScrewAxis(first_pose, z_axis));
287  auto transform = screw_axis.getTF(M_PI / 2);
288  checkVector(transform.translation(), 1.0, -1.0, 0.0); // With respect to starting position
289 
290  // Directly check the rotation matrix
291  checkVector(transform.linear().col(0), 0, 1, 0); // New x axis faces old y axis
292  checkVector(transform.linear().col(1), -1, 0, 0); // New y axis faces old -x axis
293  checkVector(transform.linear().col(2), 0, 0, 1); // z axis did not change
294 
295  // Now check linear only motion
296  affordance_primitives::ScrewAxis screw_axis_linear(MOVING_FRAME_NAME, true);
297  ASSERT_TRUE(screw_axis_linear.setScrewAxis(first_pose, z_axis));
298  transform = screw_axis_linear.getTF(M_PI / 2);
299 
300  // No rotation expected
301  EXPECT_TRUE((transform.linear() - Eigen::Matrix3d::Identity()).isZero());
302 
303  // Linear motion along z-axis by amount moved
304  checkVector(transform.translation(), 0.0, 0.0, M_PI / 2); // With respect to starting position
305 }

51 /*
52  TEST CASE:
53 
54  Screw axis at the z-axis of "task_frame" (origin = [0,0,0], axis = [0,0,1])
55  Transform from "moving_frame" to "task_frame" is:
56  Translation [2,0,0]
57  Rotate 0.5*pi about X-axis
58 
59  Pure Rotation results in:
60  [0, 0, -2*delta_theta] translational velocity
61  [0, -1*delta_theta, 0] rotational velocity
62 
63  Screw Motion (with pitch) results in:
64  [0, -1*pitch*delta_theta, -2*delta_theta] translational velocity
65  [0, -1*delta_theta, 0] rotational velocity
66 
67  Pure Translation results in:
68  [0, -1*delta_theta, 0] translational velocity
69  [0, 0, 0] rotational velocity

72 TEST(ScrewExecution, providedTF)
73 {
74  affordance_primitives::APScrewExecutor exec;
75  affordance_primitives::AffordancePrimitiveGoal ap_goal;
76  affordance_primitives::AffordancePrimitiveFeedback ap_feedback;
77 
78  // Check for bogus input case
79  ap_goal.moving_frame_source = -1;
80  bool bool_result = true;
81  ASSERT_NO_THROW(bool_result = exec.getScrewTwist(ap_goal, ap_feedback));
82  EXPECT_FALSE(bool_result);
83 
84  // Set up Screw msg
85  ap_goal.moving_frame_source = ap_goal.PROVIDED;
86  affordance_primitives::ScrewStamped screw_msg;
87  screw_msg.header.frame_id = TASK_FRAME_NAME;
88  screw_msg.axis.z = 1;
89 
90  // Check case where the transform is not (moving_frame -> task_frame)
91  affordance_primitives::TransformStamped tf_msg;
92  tf_msg.header.frame_id = MOVING_FRAME_NAME;
93  tf_msg.child_frame_id = "some_random_frame";
94  ap_goal.moving_to_task_frame = tf_msg;
95  ap_goal.screw = screw_msg;
96  bool_result = true;
97  ASSERT_NO_THROW(bool_result = exec.getScrewTwist(ap_goal, ap_feedback));
98  EXPECT_FALSE(bool_result);
99 
100  // Set up transform msg
101  tf_msg.child_frame_id = TASK_FRAME_NAME;
102  tf_msg.transform.translation.x = 2;
103  tf_msg.transform.rotation.x = 0.5 * sqrt(2);
104  tf_msg.transform.rotation.w = 0.5 * sqrt(2);
105 
106  // Check valid rotation case
107  ap_goal.moving_to_task_frame = tf_msg;
108  ap_goal.theta_dot = 0.75;
109  ap_goal.screw.is_pure_translation = false;
110  ap_goal.screw.pitch = 0;
111  bool_result = false;
112  ASSERT_NO_THROW(bool_result = exec.getScrewTwist(ap_goal, ap_feedback));
113  EXPECT_TRUE(bool_result);
114  checkVector(ap_feedback.moving_frame_twist.twist.linear, 0, 0, -2 * ap_goal.theta_dot);
115  checkVector(ap_feedback.moving_frame_twist.twist.angular, 0, -1 * ap_goal.theta_dot, 0);
116 
117  // Check valid screw motion case
118  ap_goal.screw.pitch = 1.5;
119  bool_result = false;
120  ASSERT_NO_THROW(bool_result = exec.getScrewTwist(ap_goal, ap_feedback));
121  EXPECT_TRUE(bool_result);
122  checkVector(ap_feedback.moving_frame_twist.twist.linear, 0, -1 * ap_goal.screw.pitch * ap_goal.theta_dot,
123  -2 * ap_goal.theta_dot);
124  checkVector(ap_feedback.moving_frame_twist.twist.angular, 0, -1 * ap_goal.theta_dot, 0);
125 
126  // Check valid translation case
127  ap_goal.screw.is_pure_translation = true;
128  bool_result = false;
129  ASSERT_NO_THROW(bool_result = exec.getScrewTwist(ap_goal, ap_feedback));
130  EXPECT_TRUE(bool_result);
131  checkVector(ap_feedback.moving_frame_twist.twist.linear, 0, -1 * ap_goal.theta_dot, 0);
132  checkVector(ap_feedback.moving_frame_twist.twist.angular, 0, 0, 0);
133 }

138  affordance_primitives::AffordancePrimitiveGoal ap_goal;

144 
145  // Set up Screw msg

147  affordance_primitives::ScrewStamped screw_msg;
148  screw_msg.header.frame_id = TASK_FRAME_NAME;
149  screw_msg.axis.z = 1;
150 
151  // If we look up a bogus frame, we expect no exception but a false return
152  ap_goal.moving_frame_name = "some_bogus_frame";
153  bool bool_result = true;
154  ASSERT_NO_THROW(bool_result = exec.getScrewTwist(ap_goal, ap_feedback));
155  EXPECT_FALSE(bool_result);
156

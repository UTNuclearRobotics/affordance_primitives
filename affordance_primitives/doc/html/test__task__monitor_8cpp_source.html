43 TEST(TaskMonitor, start_stop_monitor)
44 {
45  ros::NodeHandle nh;
46 
47  // Default parameters
48  affordance_primitives::APRobotParameter params;
49 
50  // Create and start monitor with no timeout

53 
54  // Wait some time, then stop monitor

71 TEST(TaskMonitor, monitor_timing)
72 {

75 
76  // Default parameters
77  affordance_primitives::APRobotParameter params;
78 
79  // Create and start monitor with timeout
80  affordance_primitives::TaskMonitor monitor(nh, "ft_topic_name");
81  ASSERT_NO_THROW(monitor.startMonitor(params, test_duration));
82 
83  // Immediately getting the result shouldn't work
84  EXPECT_FALSE(monitor.getResult().has_value());
85 
86  // Wait some time and grab result
87  ros::Duration(1.05 * test_duration).sleep();
88  std::optional<AffordancePrimitiveResult> result = monitor.getResult();
89  ASSERT_TRUE(result.has_value());
90  EXPECT_EQ(result->result, result->TIME_OUT);
91 
92  // Now test blocking call to waitForResult()
93  ASSERT_NO_THROW(monitor.startMonitor(params, test_duration));
94  AffordancePrimitiveResult result2;
95 
96  ros::Time start = ros::Time::now();
97  result2 = monitor.waitForResult();
98  EXPECT_LE((ros::Time::now() - start).toSec(), 1.05 * test_duration);
99  EXPECT_EQ(result2.result, result2.TIME_OUT);
100 }

102 TEST(TaskMonitor, monitor_ft_readings)
103 {
104  ros::NodeHandle nh;
105  ros::Publisher ft_pub = nh.advertise<affordance_primitives::WrenchStamped>("ft_topic_name", 1, true);
106 
107  const double test_duration = 0.25;
108  const double ft_pub_period = 0.005; // 200 hz
109  const size_t num_publish = test_duration / ft_pub_period;
110 
111  // Set some force limits in parameter
112  affordance_primitives::APRobotParameter params;
113  params.max_wrench.rot_x = 10;
114  params.max_wrench.rot_y = 10;
115  params.max_wrench.rot_z = 10;
116  params.max_wrench.trans_x = 100;
117  params.max_wrench.trans_y = 100;
118  params.max_wrench.trans_z = 100;
119  params.max_force = 120;
120  params.max_torque = 12;
121 
122  // Start publishing FT readings of 0
123  affordance_primitives::WrenchStamped pub_wrench;
124  std::thread t([&ft_pub, &pub_wrench, num_publish, ft_pub_period]() {
125  for (size_t i = 0; i < num_publish; ++i)
126  {
127  ft_pub.publish(pub_wrench);
128  ros::Duration(ft_pub_period).sleep();
129  }
130  });
131 
132  // Start the monitor and verify it times out
133  affordance_primitives::TaskMonitor monitor(nh, "ft_topic_name");
134  ASSERT_NO_THROW(monitor.startMonitor(params, test_duration));
135  AffordancePrimitiveResult result = monitor.waitForResult();
136  EXPECT_EQ(result.result, result.TIME_OUT);
137  t.join();
138 
139  // Now publish FT readings to violate max force limit
140  t = std::thread([&ft_pub, &pub_wrench, num_publish, ft_pub_period]() {
141  for (size_t i = 0; i < num_publish; ++i)
142  {
143  if (i == num_publish / 3)
144  {
145  pub_wrench.wrench.force.x = 99;
146  pub_wrench.wrench.force.y = 99;
147  pub_wrench.wrench.force.z = 99;
148  }
149  ft_pub.publish(pub_wrench);
150  ros::Duration(ft_pub_period).sleep();
151  }
152  });
153 
154  // Start monitor and verify FT limit is violated
155  ASSERT_NO_THROW(monitor.startMonitor(params, test_duration));
156  result = monitor.waitForResult();
157  EXPECT_EQ(result.result, result.FT_VIOLATION);
158  t.join();
159 
160  // Now publish to violate individual limits
161  pub_wrench = affordance_primitives::WrenchStamped();
162  t = std::thread([&ft_pub, &pub_wrench, num_publish, ft_pub_period]() {
163  for (size_t i = 0; i < num_publish; ++i)
164  {
165  if (i == num_publish / 3)
166  {
167  pub_wrench.wrench.torque.z = 11;
168  }
169  else
170  {
171  pub_wrench.wrench.torque.z = 9;
172  }
173  ft_pub.publish(pub_wrench);
174  ros::Duration(ft_pub_period).sleep();
175  }
176  });
177 
178  // Start monitor and verify FT limit is violated
179  ASSERT_NO_THROW(monitor.startMonitor(params, test_duration));
180  result = monitor.waitForResult();
181  EXPECT_EQ(result.result, result.FT_VIOLATION);
182  t.join();
183

187  for (size_t i = 0; i < num_publish; ++i)
